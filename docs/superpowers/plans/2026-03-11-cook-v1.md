**Goal:** Build a working Cook build system that can parse Cookfiles, transpile them to Lua, and execute recipes with dependency resolution.

**Architecture:** AST-based transpiler in Rust. Cookfile text is parsed line-by-line into an AST, validated, codegen'd to Lua, then executed via mlua with Rust-provided `cook.*` and `fs.*` APIs. Two-phase execution: register all recipes first, then resolve dependencies and run.

**Tech Stack:** Rust (2024 edition), mlua (Lua VM), clap (CLI), notify (file watching), glob (pattern matching), dotenvy (.env loading)

```rust
pub fn tokenize(source: &str) -> Result<Vec<Located<Token>>, LexError> {
    let mut tokens = Vec::new();

    for (idx, line) in source.lines().enumerate() {
        let line_num = idx + 1;
        let trimmed = line.trim();

        let token = if trimmed.is_empty() {
            Token::Blank
        } else if trimmed.starts_with('#') {
            Token::Comment(trimmed[1..].to_string())
        } else if trimmed.starts_with(">{") {
            Token::LuaBlockOpen
        } else if trimmed.starts_with('>') {
            // Trim the leading '>' and optional single space
            let code = &trimmed[1..];
            let code = code.strip_prefix(' ').unwrap_or(code);
            Token::LuaLine(code.to_string())
        } else if trimmed == "taste" {
            Token::Taste
        } else if trimmed == "end" {
            Token::RecipeEnd
        } else if trimmed.starts_with("recipe") && trimmed[6..].starts_with(|c: char| c.is_whitespace() || c == '"') {
            let rest = trimmed["recipe".len()..].trim();
            if !rest.starts_with('"') {
                return Err(LexError::MissingRecipeName { line: line_num });
            }
            let rest = &rest[1..];
            let end = rest.find('"').ok_or(LexError::UnterminatedString { line: line_num })?;
            let name = rest[..end].to_string();
            Token::RecipeHeader(name)
        } else {
            Token::Content(trimmed.to_string())
        };

        tokens.push(Located { value: token, line: line_num });
    }

    Ok(tokens)
}
```

Key fix: The `recipe` check now requires the 7th character to be whitespace or `"`, preventing false matches on words like `recipes`.

- [ ] **Step 3: Implement the parser**

The parser has three main components:
1. Top-level loop — dispatches on token type
2. Metadata expression parsing — handles `"string"` and `{"string", "string"}`
3. Lua block collection — reads raw source lines with brace-depth counting

```rust
pub fn parse(source: &str) -> Result<Cookfile, ParseError> {
    let tokens = tokenize(source)?;
    let lines: Vec<&str> = source.lines().collect();
    let mut recipes = Vec::new();
    let mut i = 0;

    while i < tokens.len() {
        match &tokens[i].value {
            Token::RecipeHeader(name) => {
                let recipe_line = tokens[i].line;
                i += 1;
                let (recipe, next_i) = parse_recipe(name.clone(), recipe_line, &tokens, i, &lines)?;
                recipes.push(recipe);
                i = next_i;
            }
            Token::Comment(_) | Token::Blank => {
                i += 1;
            }
            Token::RecipeEnd => {
                return Err(ParseError::Parse {
                    line: tokens[i].line,
                    message: "'end' outside of recipe".to_string(),
                });
            }
            other => {
                return Err(ParseError::Parse {
                    line: tokens[i].line,
                    message: format!("unexpected token at top level: {:?}", other),
                });
            }
        }
    }

    Ok(Cookfile { recipes })
}

) -> Result<(Recipe, usize), ParseError> {
    let mut metadata = RecipeMetadata::default();
    let mut steps = Vec::new();
    let mut i = start;
    let mut seen_step = false;

    while i < tokens.len() {
        match &tokens[i].value {
            Token::RecipeEnd => {
                return Ok((Recipe { name, metadata, steps, line: recipe_line }, i + 1));
            }
            Token::Comment(_) | Token::Blank => {
                i += 1;
            }
            Token::Content(text) if !seen_step => {
                // Try to parse as metadata
                if let Some((key, value_str)) = try_split_metadata(text) {
                    if KNOWN_METADATA_KEYS.contains(&key) {
                        parse_metadata_value(&mut metadata, key, value_str, tokens[i].line)?;
                        i += 1;
                        continue;
                    }
                }
                // Not metadata — it's a shell command (falls through to step handling)
                seen_step = true;
                steps.push(Step::Shell { command: text.clone(), line: tokens[i].line });
                i += 1;
            }
            Token::Content(text) => {
                seen_step = true;
                steps.push(Step::Shell { command: text.clone(), line: tokens[i].line });
                i += 1;
            }
            Token::LuaLine(code) => {
                seen_step = true;
                steps.push(Step::Lua { code: code.clone(), line: tokens[i].line });
                i += 1;
            }
            Token::LuaBlockOpen => {
                seen_step = true;
                let block_start_line = tokens[i].line;
                let (code, next_i) = collect_lua_block(block_start_line, lines)?;
                steps.push(Step::LuaBlock { code, line: block_start_line });
                // Advance token index past all lines consumed by the block
                i += 1;
                while i < tokens.len() && tokens[i].line <= next_i {
                    i += 1;
                }
            }
            Token::Taste => {
                seen_step = true;
                steps.push(Step::Taste { line: tokens[i].line });
                i += 1;
            }
            Token::RecipeHeader(_) => {
                return Err(ParseError::Parse {
                    line: tokens[i].line,
                    message: "nested recipes are not allowed".to_string(),
                });
            }
        }
    }

    Err(ParseError::Parse {
        line: recipe_line,
        message: format!("recipe \"{}\" is never closed with 'end'", name),
    })
}

/// Try to split a content line into "key" and "value_str" around "=".
/// Returns None if the line doesn't match the `ident = expr` pattern.
fn try_split_metadata(text: &str) -> Option<(&str, &str)> {
    let eq_pos = text.find('=')?;
    let key = text[..eq_pos].trim();
    // Key must be a simple identifier
    if key.is_empty() || !key.chars().all(|c| c.is_alphanumeric() || c == '_') {
        return None;
    }
    let value = text[eq_pos + 1..].trim();
    Some((key, value))
}

/// Parse a metadata value expression and store it in the RecipeMetadata.
fn parse_metadata_value(
    metadata: &mut RecipeMetadata,
    key: &str,
    value_str: &str,
    line: usize,
) -> Result<(), ParseError> {
    match key {
        "ingredients" => {
            metadata.ingredients = parse_string_table(value_str, line)?;
        }
        "serves" => {
            if value_str.starts_with('{') {
                let items = parse_string_table(value_str, line)?;
                metadata.serves = Some(Serves::Multiple(items));
            } else {
                let s = parse_string_literal(value_str, line)?;
                metadata.serves = Some(Serves::Single(s));
            }
        }
        "requires" => {
            metadata.requires = parse_string_table(value_str, line)?;
        }
        _ => unreachable!("only called with known keys"),
    }
    Ok(())
}

/// Parse a quoted string: "value" → value
fn parse_string_literal(s: &str, line: usize) -> Result<String, ParseError> {
    let s = s.trim();
    if !s.starts_with('"') || !s.ends_with('"') || s.len() < 2 {
        return Err(ParseError::Parse { line, message: "expected quoted string".to_string() });
    }
    Ok(s[1..s.len()-1].to_string())
}

/// Parse a table of strings: {"a", "b", "c"} → vec!["a", "b", "c"]
fn parse_string_table(s: &str, line: usize) -> Result<Vec<String>, ParseError> {
    let s = s.trim();
    if !s.starts_with('{') || !s.ends_with('}') {
        return Err(ParseError::Parse { line, message: "expected table literal".to_string() });
    }
    let inner = s[1..s.len()-1].trim();
    if inner.is_empty() {
        return Ok(Vec::new());
    }
    let mut items = Vec::new();
    for part in inner.split(',') {
        let item = parse_string_literal(part, line)?;
        items.push(item);
    }
    Ok(items)
}

/// Collect a Lua block starting after `>{`. Uses brace-depth counting
/// that is aware of Lua string literals and comments.
/// Returns the block content and the line number of the closing `}`.
fn collect_lua_block(start_line: usize, lines: &[&str]) -> Result<(String, usize), ParseError> {
    let mut depth = 1; // The opening `>{` counts as depth 1
    let mut block_lines = Vec::new();

    // Start from the line after `>{`
    for line_idx in start_line..lines.len() {
        let line = lines[line_idx];
        let mut chars = line.chars().peekable();
        let mut in_string: Option<char> = None; // '"' or '\''
        let mut in_long_string = false;
        let mut i = 0;
        let line_bytes = line.as_bytes();

        while i < line_bytes.len() {
            let ch = line_bytes[i] as char;

            if in_long_string {
                // Look for ]]
                if ch == ']' && i + 1 < line_bytes.len() && line_bytes[i + 1] == b']' {
                    in_long_string = false;
                    i += 2;
                    continue;
                }
                i += 1;
                continue;
            }

            if let Some(quote) = in_string {
                if ch == '\\' {
                    i += 2; // Skip escaped character
                    continue;
                }
                if ch == quote {
                    in_string = None;
                }
                i += 1;
                continue;
            }

            // Not in any string
            match ch {
                '-' if i + 1 < line_bytes.len() && line_bytes[i + 1] == b'-' => {
                    // Lua line comment — skip rest of line
                    break;
                }
                '[' if i + 1 < line_bytes.len() && line_bytes[i + 1] == b'[' => {
                    in_long_string = true;
                    i += 2;
                    continue;
                }
                '"' | '\'' => {
                    in_string = Some(ch);
                }
                '{' => depth += 1,
                '}' => {
                    depth -= 1;
                    if depth == 0 {
                        // This line contains the closing brace
                        // Add content before the closing brace if any
                        let before = line[..i].trim();
                        if !before.is_empty() {
                            block_lines.push(before.to_string());
                        }
                        return Ok((block_lines.join("\n"), line_idx + 1));
                    }
                }
                _ => {}
            }
            i += 1;
        }

        block_lines.push(line.to_string());
    }

    Err(ParseError::Parse {
        line: start_line,
        message: "unclosed Lua block (no matching '}')".to_string(),
    })
}
```

Note: The `collect_lua_block` function operates on raw source lines (not tokens) so it can accurately count braces while respecting Lua string literals, long strings, and comments. It processes each character, tracking whether it's inside a string or comment, and only counts braces in "bare" code context.

```rust
use crate::parser::ast::*;

pub fn generate(cookfile: &Cookfile) -> String {
    let mut out = String::from("-- Generated by Cook\n");

    for recipe in &cookfile.recipes {
        out.push_str(&format!("cook.recipe(\"{}\", ", recipe.name));
        out.push_str(&generate_metadata(&recipe.metadata));
        out.push_str(", function()\n");

        for step in &recipe.steps {
            match step {
                Step::Shell { command, line } => {
                    let wrapped = wrap_lua_string(command);
                    out.push_str(&format!("    cook.exec({}, {})\n", wrapped, line));
                }
                Step::Lua { code, .. } => {
                    out.push_str(&format!("    {}\n", code));
                }
                Step::LuaBlock { code, .. } => {
                    for line in code.lines() {
                        out.push_str(&format!("    {}\n", line));
                    }
                }
                Step::Taste { line } => {
                    out.push_str(&format!("    cook.taste({})\n", line));
                }
            }
        }

        out.push_str("end)\n\n");
    }

    out
}

fn generate_metadata(meta: &RecipeMetadata) -> String {
    if meta.ingredients.is_empty() && meta.serves.is_none() && meta.requires.is_empty() {
        return "{}".to_string();
    }

    let mut fields = Vec::new();

    if !meta.ingredients.is_empty() {
        let items: Vec<String> = meta.ingredients.iter().map(|s| format!("\"{}\"", s)).collect();
        fields.push(format!("ingredients = {{{}}}", items.join(", ")));
    }

    if let Some(serves) = &meta.serves {
        match serves {
            Serves::Single(s) => fields.push(format!("serves = \"{}\"", s)),
            Serves::Multiple(items) => {
                let items: Vec<String> = items.iter().map(|s| format!("\"{}\"", s)).collect();
                fields.push(format!("serves = {{{}}}", items.join(", ")));
            }
        }
    }

    if !meta.requires.is_empty() {
        let items: Vec<String> = meta.requires.iter().map(|s| format!("\"{}\"", s)).collect();
        fields.push(format!("requires = {{{}}}", items.join(", ")));
    }

    format!("{{{}}}", fields.join(", "))
}

/// Wrap a shell command in a Lua long string. Uses [[ ]] normally,
/// falls back to [=[ ]=] if the command contains ]].
fn wrap_lua_string(s: &str) -> String {
    if s.contains("]]") {
        format!("[=[{}]=]", s)
    } else {
        format!("[[{}]]", s)
    }
}
```

```rust
/// A recipe registered by Lua code via cook.recipe().
pub struct RegisteredRecipe {
    pub name: String,
    pub function: LuaRegistryKey,
}

/// Register the `cook` global table on the Lua VM.
/// Returns a shared handle to the recipe registry so the caller can
/// retrieve registered recipes after Lua execution.
pub fn register_cook_api(
    lua: &Lua,
    env_vars: &HashMap<String, String>,
    working_dir: &PathBuf,
    no_taste: bool,
) -> LuaResult<Rc<RefCell<Vec<RegisteredRecipe>>>> {
    let recipes: Rc<RefCell<Vec<RegisteredRecipe>>> = Rc::new(RefCell::new(Vec::new()));
    let cook = lua.create_table()?;

    // cook.recipe(name, metadata, fn)
    let recipes_clone = recipes.clone();
    let recipe_fn = lua.create_function(move |lua, (name, _meta, func): (String, LuaValue, LuaFunction)| {
        let key = lua.create_registry_value(func)?;
        recipes_clone.borrow_mut().push(RegisteredRecipe { name, function: key });
        Ok(())
    })?;
    cook.set("recipe", recipe_fn)?;

    // cook.exec(cmd, line) -> stdout string
    let wd = working_dir.clone();
    let env = env_vars.clone();
    let exec_fn = lua.create_function(move |_, (cmd, line): (String, usize)| {
        let mut child_env: HashMap<String, String> = std::env::vars().collect();
        // .env values override system env
        for (k, v) in &env {
            child_env.insert(k.clone(), v.clone());
        }

        let output = Command::new("/bin/sh")
            .arg("-c")
            .arg(&cmd)
            .current_dir(&wd)
            .envs(&child_env)
            .output()
            .map_err(|e| mlua::Error::runtime(format!("failed to execute: {e}")))?;

        // Pass stderr through
        if !output.stderr.is_empty() {
            eprint!("{}", String::from_utf8_lossy(&output.stderr));
        }

        if !output.status.success() {
            let code = output.status.code().unwrap_or(1);
            return Err(mlua::Error::runtime(
                format!("COOK_CMD_FAILED:{}:{}:{}", line, code, cmd)
            ));
        }

        let stdout = String::from_utf8_lossy(&output.stdout).to_string();
        Ok(stdout)
    })?;
    cook.set("exec", exec_fn)?;

    // cook.taste(line) — no-op for now (future: REPL)
    let taste_fn = lua.create_function(move |_, _line: usize| {
        if !no_taste {
            // Future: drop into interactive Lua REPL here
            // For now, just print a message
            eprintln!("[taste] breakpoint hit (REPL not yet implemented, continuing...)");
        }
        Ok(())
    })?;
    cook.set("taste", taste_fn)?;

    // cook.env — table of env vars
    let env_table = lua.create_table()?;
    for (key, value) in env_vars {
        env_table.set(key.as_str(), value.as_str())?;
    }
    cook.set("env", env_table)?;

    lua.globals().set("cook", cook)?;

    Ok(recipes)
}

/// Register the `fs` global table on the Lua VM.
pub fn register_fs_api(lua: &Lua) -> LuaResult<()> {
    let fs = lua.create_table()?;

    // fs.exists(path) -> bool
    fs.set("exists", lua.create_function(|_, path: String| {
        Ok(std::path::Path::new(&path).exists())
    })?)?;

    // fs.size(path) -> number
    fs.set("size", lua.create_function(|_, path: String| {
        let meta = std::fs::metadata(&path)
            .map_err(|e| mlua::Error::runtime(format!("fs.size: {e}")))?;
        Ok(meta.len())
    })?)?;

    // fs.read(path) -> string
    fs.set("read", lua.create_function(|_, path: String| {
        let content = std::fs::read_to_string(&path)
            .map_err(|e| mlua::Error::runtime(format!("fs.read: {e}")))?;
        Ok(content)
    })?)?;

    // fs.glob(pattern) -> table of strings
    fs.set("glob", lua.create_function(|lua, pattern: String| {
        let paths: Vec<String> = glob::glob(&pattern)
            .map_err(|e| mlua::Error::runtime(format!("fs.glob: {e}")))?
            .filter_map(|p| p.ok())
            .map(|p| p.to_string_lossy().to_string())
            .collect();
        let table = lua.create_table()?;
        for (i, path) in paths.iter().enumerate() {
            table.set(i + 1, path.as_str())?;
        }
        Ok(table)
    })?)?;

    // fs.mtime(path) -> number (seconds since epoch)
    fs.set("mtime", lua.create_function(|_, path: String| {
        let meta = std::fs::metadata(&path)
            .map_err(|e| mlua::Error::runtime(format!("fs.mtime: {e}")))?;
        let mtime = meta.modified()
            .map_err(|e| mlua::Error::runtime(format!("fs.mtime: {e}")))?;
        let duration = mtime.duration_since(std::time::UNIX_EPOCH)
            .map_err(|e| mlua::Error::runtime(format!("fs.mtime: {e}")))?;
        Ok(duration.as_secs_f64())
    })?)?;

    lua.globals().set("fs", fs)?;
    Ok(())
}
```

```rust
    quiet: bool,
}

impl Runtime {
    pub fn new(working_dir: PathBuf, env_vars: HashMap<String, String>) -> Self {
        Self { working_dir, env_vars, no_taste: false, quiet: false }
    }

    pub fn set_quiet(&mut self, quiet: bool) { self.quiet = quiet; }

        Ok(lua)
    }

    pub fn execute_recipe(&self, lua_source: &str, recipe_name: &str) -> Result<(), RuntimeError> {
        let lua = self.create_vm()?;
        let recipes = register_cook_api(&lua, &self.env_vars, &self.working_dir, self.no_taste)?;
        register_fs_api(&lua)?;

        lua.load(lua_source).exec()?;

        let registry = recipes.borrow();
        let recipe = registry.iter().find(|r| r.name == recipe_name)
            .ok_or_else(|| RuntimeError::RecipeNotFound(recipe_name.to_string()))?;

        let func: LuaFunction = lua.registry_value(&recipe.function)?;
        match func.call::<()>(()) {
            Ok(()) => Ok(()),
            Err(mlua::Error::RuntimeError(msg)) if msg.contains("COOK_CMD_FAILED:") => {
                // Parse the structured error from cook.exec
                let parts: Vec<&str> = msg.split("COOK_CMD_FAILED:").nth(1)
                    .unwrap_or("0:1:unknown").splitn(3, ':').collect();
                let line = parts.first().and_then(|s| s.parse().ok()).unwrap_or(0);
                let code = parts.get(1).and_then(|s| s.parse().ok()).unwrap_or(1);
                let command = parts.get(2).unwrap_or(&"unknown").to_string();
                Err(RuntimeError::CommandFailed { command, line, code })
            }
            Err(e) => Err(RuntimeError::Lua(e)),
        }
    }

    pub fn list_recipes(&self, lua_source: &str) -> Result<Vec<String>, RuntimeError> {
        let lua = self.create_vm()?;
        let recipes = register_cook_api(&lua, &self.env_vars, &self.working_dir, self.no_taste)?;
        register_fs_api(&lua)?;

        lua.load(lua_source).exec()?;

        let registry = recipes.borrow();
        Ok(registry.iter().map(|r| r.name.clone()).collect())
    }
```

- [ ] **Step 3: Implement the dependency graph**

The algorithm:
1. Build an adjacency map: for each recipe, collect all dependencies:
   - Explicit: `requires` entries → direct recipe name deps
   - Implicit: for each ingredient string, check if any other recipe `serves` that exact string
2. Starting from the target, DFS to collect all needed recipes
3. Topological sort with cycle detection using three states: unvisited, visiting (in current DFS path), visited

```rust
pub fn topological_sort(
    recipes: &HashMap<String, RecipeInfo>,
    target: &str,
) -> Result<Vec<String>, GraphError> {
    if !recipes.contains_key(target) {
        return Err(GraphError::UnknownRecipe(target.to_string()));
    }

    // Build serves → recipe name lookup
    let mut serves_map: HashMap<&str, &str> = HashMap::new();
    for (name, info) in recipes {
        for path in &info.serves {
            serves_map.insert(path.as_str(), name.as_str());
        }
    }

    // Build adjacency: recipe → set of recipes it depends on
    let mut deps: HashMap<&str, HashSet<&str>> = HashMap::new();
    for (name, info) in recipes {
        let mut recipe_deps = HashSet::new();
        for req in &info.requires {
            if !recipes.contains_key(req.as_str()) {
                return Err(GraphError::UnknownRecipe(req.clone()));
            }
            recipe_deps.insert(req.as_str());
        }
        for ingredient in &info.ingredients {
            if let Some(&provider) = serves_map.get(ingredient.as_str()) {
                if provider != name.as_str() {
                    recipe_deps.insert(provider);
                }
            }
        }
        deps.insert(name.as_str(), recipe_deps);
    }

    // DFS topological sort from target
    #[derive(Clone, Copy, PartialEq)]
    enum State { Unvisited, Visiting, Visited }

    let mut states: HashMap<&str, State> = recipes.keys().map(|k| (k.as_str(), State::Unvisited)).collect();
    let mut order = Vec::new();

    fn visit<'a>(
        node: &'a str,
        deps: &HashMap<&'a str, HashSet<&'a str>>,
        states: &mut HashMap<&'a str, State>,
        order: &mut Vec<String>,
    ) -> Result<(), GraphError> {
        match states.get(node) {
            Some(State::Visited) => return Ok(()),
            Some(State::Visiting) => return Err(GraphError::CycleDetected(node.to_string())),
            _ => {}
        }
        states.insert(node, State::Visiting);
        if let Some(node_deps) = deps.get(node) {
            for &dep in node_deps {
                visit(dep, deps, states, order)?;
            }
        }
        states.insert(node, State::Visited);
        order.push(node.to_string());
        Ok(())
    }

    visit(target, &deps, &mut states, &mut order)?;
    Ok(order)
}
```

```rust
    #[arg(short, long, global = true)]
    pub quiet: bool,

    /// Skip taste breakpoints
    #[arg(long, global = true)]
    pub no_taste: bool,
}

#[derive(Subcommand)]
pub enum Command {
    /// Run a recipe (explicit subcommand form)
    Run {
        /// Recipe to run
        #[arg(default_value = "build")]
        recipe: String,
    },
    /// Watch ingredients and re-run on change
    Serve {
        /// Recipe to serve
        #[arg(default_value = "build")]
        recipe: String,
    },
    /// List all recipes
    Menu,
    /// Generate a starter Cookfile
    Init,
}

#[derive(Debug)]
enum CookError {
    ParseError(String),
    RecipeNotFound(String),
    CommandFailed(String),
    Other(String),
}

impl CookError {
    fn exit_code(&self) -> i32 {
        match self {
            CookError::CommandFailed(_) => 1,
            CookError::ParseError(_) => 2,
            CookError::RecipeNotFound(_) => 3,
            CookError::Other(_) => 1,
        }
    }
}

impl std::fmt::Display for CookError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            CookError::ParseError(msg) => write!(f, "parse error: {msg}"),
            CookError::RecipeNotFound(name) => write!(f, "recipe not found: {name}"),
            CookError::CommandFailed(msg) => write!(f, "{msg}"),
            CookError::Other(msg) => write!(f, "{msg}"),
        }
    }
}

pub fn run() -> Result<(), Box<dyn std::error::Error>> {
    let cli = Cli::parse();

    let result = match &cli.command {
        Some(Command::Init) => cmd_init(),
        Some(Command::Menu) => cmd_menu(&cli),
        Some(Command::Serve { recipe }) => cmd_serve(&cli, recipe),
        Some(Command::Run { recipe }) => cmd_run(&cli, recipe),
        None => cmd_run(&cli, "build"),
    };

    match result {
        Ok(()) => Ok(()),
        Err(e) => {
            eprintln!("cook: {e}");
            std::process::exit(e.exit_code());
        }
    }
}

fn read_and_parse(cli: &Cli) -> Result<(crate::parser::ast::Cookfile, String), CookError> {
    let source = fs::read_to_string(&cli.file)
        .map_err(|e| CookError::Other(format!("cannot read {}: {e}", cli.file.display())))?;

    let cookfile = parser::parse(&source)
        .map_err(|e| CookError::ParseError(e.to_string()))?;

    let lua_source = codegen::generate(&cookfile);

    Ok((cookfile, lua_source))
}

fn cmd_run(cli: &Cli, recipe_name: &str) -> Result<(), CookError> {
    let (cookfile, lua_source) = read_and_parse(cli)?;

    if cli.plate {
        println!("{lua_source}");
        return Ok(());
    }

    let cookfile_dir = cli.file.parent().unwrap_or(std::path::Path::new("."));
    let env_vars = load_env(cookfile_dir);

    // Resolve dependency order
    let order = analyzer::resolve_execution_order(&cookfile, recipe_name)
        .map_err(|e| match e {
            analyzer::graph::GraphError::UnknownRecipe(name) => CookError::RecipeNotFound(name),
            analyzer::graph::GraphError::CycleDetected(name) => CookError::Other(format!("dependency cycle involving: {name}")),
        })?;

    // Execute each recipe in dependency order
    let mut rt = Runtime::new(
        cookfile_dir.to_path_buf(),
        env_vars,
    );
    rt.set_no_taste(cli.no_taste);
    rt.set_quiet(cli.quiet);

    for name in &order {
        if !cli.quiet {
            eprintln!("cook: running recipe '{name}'");
        }
        rt.execute_recipe(&lua_source, name)
            .map_err(|e| match e {
                crate::runtime::RuntimeError::RecipeNotFound(name) => CookError::RecipeNotFound(name),
                crate::runtime::RuntimeError::CommandFailed { command, line, code } => {
                    CookError::CommandFailed(format!("Cookfile:{line}: command failed (exit {code}): {command}"))
                }
                crate::runtime::RuntimeError::Lua(e) => CookError::Other(format!("lua error: {e}")),
            })?;
    }

    Ok(())
}

fn cmd_menu(cli: &Cli) -> Result<(), CookError> {
    let (cookfile, _) = read_and_parse(cli)?;

    for recipe in &cookfile.recipes {
        let mut desc = format!("  {}", recipe.name);
        if !recipe.metadata.ingredients.is_empty() {
            desc.push_str(&format!("  ingredients: {:?}", recipe.metadata.ingredients));
        }
        if let Some(serves) = &recipe.metadata.serves {
            match serves {
                crate::parser::ast::Serves::Single(s) => desc.push_str(&format!("  serves: {s}")),
                crate::parser::ast::Serves::Multiple(v) => desc.push_str(&format!("  serves: {:?}", v)),
            }
        }
        if !recipe.metadata.requires.is_empty() {
            desc.push_str(&format!("  requires: {:?}", recipe.metadata.requires));
        }
        println!("{desc}");
    }

    Ok(())
}

fn cmd_init() -> Result<(), CookError> {
    let path = std::path::Path::new("Cookfile");
    if path.exists() {
        return Err(CookError::Other("Cookfile already exists".to_string()));
    }
    fs::write(path, r#"recipe "build"

    # serves = "bin/app"
    echo "Hello from Cook!"
end
"#).map_err(|e| CookError::Other(format!("failed to write Cookfile: {e}")))?;
    println!("Created Cookfile");
    Ok(())
}
```

```rust
use notify::{Watcher, RecommendedWatcher, RecursiveMode, Event, EventKind};
use std::path::{Path, PathBuf};
use std::sync::mpsc;
use std::time::{Duration, Instant};

pub struct CookWatcher {
    pub globs: Vec<String>,
    pub cookfile_path: PathBuf,
}

impl CookWatcher {
    pub fn new(globs: Vec<String>, cookfile_path: PathBuf) -> Self {
        Self { globs, cookfile_path }
    }

    /// Collect all ingredient globs from a set of recipes (by name).
    pub fn collect_globs_for_recipes(
        cookfile: &crate::parser::ast::Cookfile,
        recipe_names: &[String],
    ) -> Vec<String> {
        let mut globs = Vec::new();
        for recipe in &cookfile.recipes {
            if recipe_names.contains(&recipe.name) {
                globs.extend(recipe.metadata.ingredients.clone());
            }
        }
        globs
    }

    /// Check if a changed path matches any of the ingredient globs.
    fn matches_any_glob(&self, path: &Path) -> bool {
        let path_str = path.to_string_lossy();
        for pattern in &self.globs {
            if let Ok(glob_pattern) = glob::Pattern::new(pattern) {
                if glob_pattern.matches(&path_str) {
                    return true;
                }
            }
        }
        false
    }

    /// Start watching. Calls `on_change` whenever a relevant file changes.
    /// Blocks the current thread. Returns only on error.
    pub fn watch<F>(&self, on_change: F) -> Result<(), Box<dyn std::error::Error>>
    where
        F: Fn(bool) -> Result<(), Box<dyn std::error::Error>>,
        // bool = true if Cookfile changed (needs re-parse), false otherwise
    {
        let (tx, rx) = mpsc::channel();

        let mut watcher = RecommendedWatcher::new(
            move |res: Result<Event, notify::Error>| {
                if let Ok(event) = res {
                    let _ = tx.send(event);
                }
            },
            notify::Config::default(),
        )?;

        // Watch directories containing glob patterns
        let mut watched_dirs = std::collections::HashSet::new();
        for pattern in &self.globs {
            // Watch the parent directory of each glob pattern
            let dir = Path::new(pattern)
                .parent()
                .unwrap_or(Path::new("."));
            if watched_dirs.insert(dir.to_path_buf()) && dir.exists() {
                watcher.watch(dir, RecursiveMode::Recursive)?;
            }
        }

        // Also watch the Cookfile
        if let Some(cookfile_dir) = self.cookfile_path.parent() {
            watcher.watch(cookfile_dir, RecursiveMode::NonRecursive)?;
        }

        let debounce = Duration::from_millis(200);
        let mut last_trigger = Instant::now() - debounce;

        loop {
            match rx.recv() {
                Ok(event) => {
                    if Instant::now().duration_since(last_trigger) < debounce {
                        continue;
                    }

                    let cookfile_changed = event.paths.iter()
                        .any(|p| p == &self.cookfile_path);

                    let relevant = cookfile_changed || event.paths.iter()
                        .any(|p| self.matches_any_glob(p));

                    if relevant {
                        last_trigger = Instant::now();
                        if let Err(e) = on_change(cookfile_changed) {
                            eprintln!("cook serve: rebuild failed: {e}");
                        }
                    }
                }
                Err(e) => {
                    return Err(format!("watch error: {e}").into());
                }
            }
        }
    }
}
```

```rust
fn cmd_serve(cli: &Cli, recipe_name: &str) -> Result<(), CookError> {
    let (cookfile, lua_source) = read_and_parse(cli)?;

    let order = analyzer::resolve_execution_order(&cookfile, recipe_name)
        .map_err(|e| match e {
            analyzer::graph::GraphError::UnknownRecipe(name) => CookError::RecipeNotFound(name),
            analyzer::graph::GraphError::CycleDetected(name) => CookError::Other(format!("dependency cycle involving: {name}")),
        })?;

    let globs = crate::watcher::CookWatcher::collect_globs_for_recipes(&cookfile, &order);
    if globs.is_empty() {
        return Err(CookError::Other("nothing to watch: no recipes in the chain have ingredients".to_string()));
    }

    let cookfile_path = std::fs::canonicalize(&cli.file)
        .map_err(|e| CookError::Other(format!("cannot resolve Cookfile path: {e}")))?;
    let watcher = crate::watcher::CookWatcher::new(globs, cookfile_path);

    // Run initial build
    eprintln!("cook serve: initial build...");
    let _ = cmd_run(cli, recipe_name);

    eprintln!("cook serve: watching for changes...");
    watcher.watch(|cookfile_changed| {
        if cookfile_changed {
            eprintln!("cook serve: Cookfile changed, rebuilding...");
        }
        // Re-run the full pipeline (re-parse if Cookfile changed)
        cmd_run(cli, recipe_name).map_err(|e| e.to_string())?;
        Ok(())
    }).map_err(|e| CookError::Other(e.to_string()))?;

    Ok(())
}
```

Note: Recipes with no ingredients that are dependencies of the target only run during the initial build and re-triggered builds — they are not independently watched since they have no ingredients to trigger changes.
