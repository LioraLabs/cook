# cc Module M0 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Promote the C/C++ build module from per-example vendoring to a blessed Standard module — published as `cook_cc-0.1.0-1` on `rocks.usecook.com`, leveraging `lua-cjson` + `lpeg`, contracted by a normative §9 Standard chapter, with end-to-end conformance and three migrated examples.

**Architecture:** Cross-repo M0 split into 6 sequenced slices (M0.0–M0.6) plus an acceptance gate (M0.7). The `cook` repo carries the engine fix, Standard chapter, examples migration, and conformance fixtures. The `cook-modules` monorepo carries the rewritten module and its publish flow. Source-of-truth design: `docs/superpowers/specs/2026-05-11-cpp-module-m0-revision-2-design.md`.

**Tech Stack:** Rust (mlua, cook-register, cook-luaotp), Lua 5.4, `lua-cjson ~> 2.1`, `lpeg ~> 1.0`, `busted` (Lua unit-test framework), LuaRocks 3.11, Cook CLI, MDX (Astro Starlight) for the Standard.

Write `cook_cc/spec/cook_stub.lua`:

```lua
-- Minimal stand-ins for the cook-engine-provided globals so busted can
-- exercise cook_cc submodules in isolation. Each spec resets the state.

local M = {}

local cache_store = {}
local export_store = {}
local added_units = {}
local sh_handlers = {}   -- map[string] -> function(cmd) -> stdout

function M.reset()
    cache_store = {}
    export_store = {}
    added_units = {}
    sh_handlers = {}
end

function M.set_sh_handler(prefix, handler)
    sh_handlers[prefix] = handler
end

function M.added_units()
    return added_units
end

function M.install()
    _G.cook = {
        env = setmetatable({}, { __index = function() return nil end }),
        platform = { os = "linux" },
        cache = {
            get = function(k) return cache_store[k] end,
            set = function(k, v) cache_store[k] = v end,
        },
        export = function(name, info) export_store[name] = info end,
        import = function(name) return export_store[name] end,
        add_unit = function(u) added_units[#added_units + 1] = u end,
        sh = function(cmd)
            for prefix, fn in pairs(sh_handlers) do
                if cmd:sub(1, #prefix) == prefix then return fn(cmd) end
            end
            error("[cook_stub] unhandled sh: " .. cmd)
        end,
    }

    _G.fs = {
        exists = function(p) return sh_handlers.__exists and sh_handlers.__exists(p) or true end,
        read   = function(p) return (sh_handlers.__read or function() return "" end)(p) end,
        write  = function(p, content)
            added_units[#added_units + 1] = { kind = "fs.write", path = p, content = content }
        end,
        mkdir_p = function() end,
        glob    = function(_) return {} end,
    }

    _G.path = {
        stem = function(p) return p:match("([^/]+)%.[^.]+$") end,
        dir  = function(p) return p:match("(.+)/[^/]+$") or "." end,
    }
end

return M
```

local M = {}

-- File-local state. Per-VM; rehydrated from cook.cache by init.lua.
local state = {

    default_standard = nil,    -- e.g. "c++17"
    warnings         = "default",
    defaults         = {},     -- additive-merge target defaults
}

local function compiler_present(cxx)
    local ok = pcall(function() cook.sh("command -v " .. cxx) end)
    return ok
end

local function detect_compiler()
    for _, candidate in ipairs({
        { cxx = "g++",     cc = "gcc"   },
        { cxx = "clang++", cc = "clang" },
    }) do
        if compiler_present(candidate.cxx) then return candidate end
    end
    error("[cc.toolchain] no C/C++ compiler on PATH (tried g++, clang++)")
end

function M.rehydrate()
    local cached = cook.cache.get("compiler")
    if cached and compiler_present(cached.cxx) then
        state.compiler = cached
        return
    end
    state.compiler = detect_compiler()
    cook.cache.set("compiler", state.compiler)
end

function M.set(opts)
    opts = opts or {}
    if opts.compiler then
        local cxx = opts.compiler
        local cc
        if cxx:match("clang") then cc = "clang"
        elseif cxx:match("g%+%+") then cc = "gcc"
        else cc = "cc" end
        state.compiler = { cxx = cxx, cc = cc }
    end
    if opts.standard then state.default_standard = opts.standard end
    if opts.warnings then state.warnings = opts.warnings end
end

local function append_list(dst, src)
    if not src then return end
    for _, v in ipairs(src) do dst[#dst + 1] = v end
end

function M.merge_defaults(opts)
    opts = opts or {}
    state.defaults.defines     = state.defaults.defines     or {}
    state.defaults.includes    = state.defaults.includes    or {}
    state.defaults.system_libs = state.defaults.system_libs or {}
    append_list(state.defaults.defines,     opts.defines)
    append_list(state.defaults.includes,    opts.includes)
    append_list(state.defaults.system_libs, opts.system_libs)
    if opts.extra_cflags then
        state.defaults.extra_cflags = (state.defaults.extra_cflags or "") .. " " .. opts.extra_cflags
    end
    if opts.extra_ldflags then
        state.defaults.extra_ldflags = (state.defaults.extra_ldflags or "") .. " " .. opts.extra_ldflags
    end
    if opts.standard then state.default_standard = opts.standard end
    if opts.warnings then state.warnings = opts.warnings end
end

function M.get_compiler()         return state.compiler         end
function M.get_default_standard() return state.default_standard end
function M.get_warnings()         return state.warnings         end
function M.get_defaults()         return state.defaults         end

function M.warning_flags(override)
    local w = override or state.warnings
    if w == "default" then return "-Wall"
    elseif w == "strict" then return "-Wall -Wextra -Wpedantic"
    elseif w == "none" then return ""
    else return w end
end

return M

local M = {}

local function add_unique(dst, src, seen)
    if not src then return end
    for _, v in ipairs(src) do
        if not seen[v] then
            seen[v] = true
            dst[#dst + 1] = v
        end
    end
end

function M.resolve_links(links)
    local merged = {
        includes      = {},
        defines       = {},
        system_libs   = {},
        extra_ldflags = "",
    }
    local seen_inc, seen_def, seen_lib = {}, {}, {}
    local visited = {}

    local function walk(name)
        if visited[name] then return end
        visited[name] = true
        local info = cook.import(name)
        if not info then return end
        add_unique(merged.includes,    info.includes,    seen_inc)
        add_unique(merged.defines,     info.defines,     seen_def)
        add_unique(merged.system_libs, info.system_libs, seen_lib)
        if info.extra_ldflags and info.extra_ldflags ~= "" then
            if merged.extra_ldflags ~= "" then
                merged.extra_ldflags = merged.extra_ldflags .. " " .. info.extra_ldflags
            else
                merged.extra_ldflags = info.extra_ldflags
            end
        end
        for _, child in ipairs(info.links or {}) do walk(child) end
    end

    for _, name in ipairs(links or {}) do walk(name) end
    return merged
end

return M

local toolchain = require("cook_cc.toolchain")

local M = {}

local function is_c(source) return source:match("%.[cC]$") ~= nil end

local function compiler_for(source)
    local c = toolchain.get_compiler()
    return is_c(source) and c.cc or c.cxx
end

function M.compile(source, opts)
    opts = opts or {}
    if not fs.exists(source) then
        error("[cc.compile] source file not found: " .. source, 2)
    end
    local target_name = opts.target_name or "default"
    local stem = path.stem(source)
    local obj_dir = "build/obj/" .. target_name
    local obj_out = opts.output or (obj_dir .. "/" .. stem .. ".o")
    local dep_dir = ".cook/deps/" .. target_name
    local dep_file = dep_dir .. "/" .. stem .. ".d"

    fs.mkdir_p(obj_dir)
    fs.mkdir_p(dep_dir)

    local flags = { "-c", "-MMD", "-MF", dep_file }

    local std = opts.standard or toolchain.get_default_standard()
    if std and not is_c(source) then
        flags[#flags + 1] = "-std=" .. std
    end

    for _, inc in ipairs(opts.includes or {}) do flags[#flags + 1] = "-I" .. inc end
    for _, def in ipairs(opts.defines  or {}) do flags[#flags + 1] = "-D" .. def end

    local wflags = toolchain.warning_flags(opts.warnings)
    if wflags ~= "" then flags[#flags + 1] = wflags end

    if opts.fpic then flags[#flags + 1] = "-fPIC" end
    if opts.extra_cflags and opts.extra_cflags ~= "" then
        flags[#flags + 1] = opts.extra_cflags
    end

    local cmd = compiler_for(source) .. " " .. table.concat(flags, " ")
        .. " " .. source .. " -o " .. obj_out

    cook.add_unit({
        inputs = { source },
        output = obj_out,
        command = cmd,
        discovered_inputs = { from = dep_file, format = "make" },
    })

    return obj_out
end

function M.archive(objects, output)
    fs.mkdir_p(path.dir(output))
    local cmd = "ar rcs " .. output .. " " .. table.concat(objects, " ")
    cook.add_unit({
        inputs = objects,
        output = output,
        command = cmd,
    })
    return output
end

function M.link(objects, output, opts)
    opts = opts or {}
    fs.mkdir_p(path.dir(output))
    local cxx = toolchain.get_compiler().cxx
    local parts = { cxx, table.concat(objects, " "), "-o", output }
    for _, lib in ipairs(opts.system_libs or {}) do parts[#parts + 1] = "-l" .. lib end
    if opts.extra_ldflags and opts.extra_ldflags ~= "" then
        parts[#parts + 1] = opts.extra_ldflags
    end
    if opts.shared then parts[#parts + 1] = "-shared" end

    -- pad for matchable spaces in tests
    local cmd = table.concat(parts, " ") .. " "
    cmd = cxx .. " " .. table.concat(objects, " ") .. " -o " .. output
    for _, lib in ipairs(opts.system_libs or {}) do cmd = cmd .. " -l" .. lib end
    if opts.extra_ldflags and opts.extra_ldflags ~= "" then
        cmd = cmd .. " " .. opts.extra_ldflags
    end
    if opts.shared then cmd = cmd .. " -shared" end
    -- trailing space so the " -lname " assertions match cleanly at end-of-string too
    cmd = cmd .. " "

    cook.add_unit({
        inputs = objects,
        output = output,
        command = cmd,
    })
    return output
end

return M

local cc         = require("cook_cc.cc")
local toolchain  = require("cook_cc.toolchain")
local transitive = require("cook_cc.transitive")

local M = {}

local function register_known(name)
    local known = cook.cache.get("known_targets") or {}
    for _, n in ipairs(known) do if n == name then return end end
    known[#known + 1] = name
    cook.cache.set("known_targets", known)
end

local function gather_sources(opts)
    local sources = opts.sources or {}
    if (#sources == 0) and opts.dir then
        for _, ext in ipairs({ "*.cpp", "*.c", "*.cc", "*.cxx" }) do
            for _, m in ipairs(fs.glob(opts.dir .. "**/" .. ext)) do
                sources[#sources + 1] = m
            end
        end
    end
    return sources
end

local function build_opts(opts, kind)
    opts = opts or {}
    local d = toolchain.get_defaults()
    local merged_includes = {}
    local merged_defines  = {}
    local merged_libs     = {}
    for _, v in ipairs(d.includes    or {}) do merged_includes[#merged_includes + 1] = v end
    for _, v in ipairs(opts.includes or {}) do merged_includes[#merged_includes + 1] = v end
    for _, v in ipairs(d.defines     or {}) do merged_defines [#merged_defines  + 1] = v end
    for _, v in ipairs(opts.defines  or {}) do merged_defines [#merged_defines  + 1] = v end
    for _, v in ipairs(d.system_libs    or {}) do merged_libs[#merged_libs + 1] = v end
    for _, v in ipairs(opts.system_libs or {}) do merged_libs[#merged_libs + 1] = v end
    return {
        includes      = merged_includes,
        defines       = merged_defines,
        system_libs   = merged_libs,
        standard      = opts.standard,
        warnings      = opts.warnings,
        extra_cflags  = opts.extra_cflags,
        extra_ldflags = opts.extra_ldflags,
        export_includes = opts.export_includes,
        links         = opts.links or {},
        fpic          = (kind == "shared"),
    }
end

local function record_export(name, sources, b)
    cook.export(name, {
        includes      = b.export_includes or b.includes,
        defines       = b.defines,
        system_libs   = b.system_libs,
        extra_ldflags = b.extra_ldflags or "",
        links         = b.links,
        compile_info  = {
            sources  = sources,
            includes = b.includes,
            defines  = b.defines,
            standard = b.standard,
            compiler = toolchain.get_compiler() and toolchain.get_compiler().cxx,
        },
    })
end

local function compile_all(name, sources, b)
    local objs = {}
    for _, src in ipairs(sources) do
        objs[#objs + 1] = cc.compile(src, {
            target_name  = name,
            includes     = b.includes,
            defines      = b.defines,
            standard     = b.standard,
            warnings     = b.warnings,
            extra_cflags = b.extra_cflags,
            fpic         = b.fpic,
        })
    end
    return objs
end

function M.bin(name, opts)
    local b = build_opts(opts, "bin")
    local sources = gather_sources(opts or {})
    if #sources == 0 then
        error("[cc.bin] no sources found for target '" .. name .. "'", 2)
    end
    register_known(name)
    record_export(name, sources, b)
    local objs = compile_all(name, sources, b)
    local merged = transitive.resolve_links(b.links)
    cc.link(objs, "build/bin/" .. name, {
        system_libs   = merged.system_libs[1] and merged.system_libs or b.system_libs,
        extra_ldflags = (merged.extra_ldflags ~= "" and merged.extra_ldflags) or b.extra_ldflags,
    })
    return name

        extra_ldflags = (merged.extra_ldflags ~= "" and merged.extra_ldflags) or b.extra_ldflags,

local lpeg = require("lpeg")
local P, R, S, C, Cg, Ct, V = lpeg.P, lpeg.R, lpeg.S, lpeg.C, lpeg.Cg, lpeg.Ct, lpeg.V

local M = {}

-- Grammar: parses a pkg-config-output line into structured fields.
local space    = S(" \t")^0
local non_space = (P(1) - S(" \t"))^1
local include  = P("-I") * C(non_space)
local libdir   = P("-L") * C(non_space)
local syslib   = P("-l") * C(non_space)
local define   = P("-D") * C(non_space)
local framework = P("-framework") * S(" \t")^1 * C(non_space)
local other    = C(non_space)

local token = framework / function(v) return { kind = "framework", value = v } end
            + include   / function(v) return { kind = "include",   value = v } end
            + libdir    / function(v) return { kind = "libdir",    value = v } end
            + syslib    / function(v) return { kind = "syslib",    value = v } end
            + define    / function(v) return { kind = "define",    value = v } end
            + other     / function(v) return { kind = "other",     value = v } end

local line = Ct((space * token)^0 * space)

local function parse_tokens(s)
    return line:match(s or "") or {}
end

local function shell_chomp(s) return (s or ""):gsub("%s+$", "") end

local function try_sh(cmd)
    local ok, out = pcall(cook.sh, cmd)
    return ok, ok and shell_chomp(out) or nil
end

local function blank_result()
    return {
        found        = false,
        cflags       = "",
        libs         = "",
        system_libs  = {},
        include_dirs = {},
        lib_dirs     = {},
        frameworks   = {},
        version      = nil,
    }
end

function M.find(name, _opts)
    local cache_key = "pkg:" .. name
    local cached = cook.cache.get(cache_key)
    if cached then return cached end

    local ok = try_sh("pkg-config --exists " .. name)
    if not ok then
        local r = blank_result()
        cook.cache.set(cache_key, r)
        return r
    end
    local _, cflags = try_sh("pkg-config --cflags " .. name)
    local _, libs   = try_sh("pkg-config --libs "   .. name)

    local result = blank_result()
    result.found  = true
    result.cflags = cflags or ""
    result.libs   = libs   or ""

    local function bucket(toks)
        for _, t in ipairs(toks) do
            if     t.kind == "include"   then result.include_dirs[#result.include_dirs + 1] = t.value
            elseif t.kind == "libdir"    then result.lib_dirs[#result.lib_dirs + 1] = t.value
            elseif t.kind == "syslib"    then result.system_libs[#result.system_libs + 1] = t.value
            elseif t.kind == "framework" then result.frameworks[#result.frameworks + 1] = t.value
            end
        end
    end
    bucket(parse_tokens(result.cflags))
    bucket(parse_tokens(result.libs))

    cook.cache.set(cache_key, result)
    return result
end

return M
```

local cjson = require("cjson")
local toolchain = require("cook_cc.toolchain")

local M = {}

local function is_c(source) return source:match("%.[cC]$") ~= nil end

local function shell_chomp(s) return (s or ""):gsub("%s+$", "") end

local function format_command(src, ci)
    local compiler = ci.compiler or "g++"
    if is_c(src) then
        -- map cxx → cc (g++→gcc, clang++→clang); fallback "cc"
        if compiler:match("g%+%+")    then compiler = "gcc"
        elseif compiler:match("clang") then compiler = "clang"
        else compiler = "cc" end
    end
    local flags = { "-c" }
    if ci.standard and not is_c(src) then
        flags[#flags + 1] = "-std=" .. ci.standard
    end
    for _, inc in ipairs(ci.includes or {}) do flags[#flags + 1] = "-I" .. inc end
    for _, def in ipairs(ci.defines  or {}) do flags[#flags + 1] = "-D" .. def end
    local stem = src:match("([^/]+)%.[^.]+$") or src
    return compiler .. " " .. table.concat(flags, " ") .. " " .. src
        .. " -o build/obj/" .. (ci.target_name or "default") .. "/" .. stem .. ".o"
end

local function write_with_directory(entries, wd)
    local out = { "[" }
    for i, e in ipairs(entries) do
        out[#out + 1] = (i == 1 and "" or ",")
        out[#out + 1] = cjson.encode(e)
    end
    out[#out + 1] = "]\n"
    fs.write("compile_commands.json", table.concat(out))
end

function M.write()
    local targets = cook.cache.get("known_targets") or {}
    local entries = {}
    if #targets == 0 then
        fs.write("compile_commands.json", "[]\n")
        return
    end
    local wd = shell_chomp(cook.sh("pwd"))
    for _, name in ipairs(targets) do
        local info = cook.import(name)
        if info and info.compile_info then
            local ci = info.compile_info
            ci.target_name = name
            for _, src in ipairs(ci.sources or {}) do
                entries[#entries + 1] = {
                    directory = wd,
                    command   = format_command(src, ci),
                    file      = src,
                }
            end
        end
    end
    -- Encode the full array as JSON so cjson.decode round-trips.
    fs.write("compile_commands.json", cjson.encode(entries) .. "\n")
end

return M
```

## 9.1 Scope and bootstrap

This chapter normatively specifies the public surface of blessed Cook modules — modules curated by the Cook project and distributed through the official LuaRocks-backed registry (`rocks.usecook.com`). A module's presence as a numbered subsection of this chapter is what makes it "blessed": a conformance-checkable contract that any implementation of the module must honour.

The Standard governs the contract, not the reference implementation. The published rock is one implementation that meets the contract; a reimplementation in another language layer must produce semantically equivalent behaviour for every public surface specified here.

### 9.1.1 Bootstrap

A conforming project obtains a blessed module by listing it under `[modules]` in `cook.toml` and running `cook modules install`. A project MAY hand-vendor a copy of a blessed module by placing `cook_modules/<name>.lua` or `cook_modules/<name>/init.lua` at the top of the project's `cook_modules/` directory. Per Standard §7, hand-vendored top-level paths take precedence over LuaRocks-installed `share/lua/5.4/` paths. ### 9.2.1 Synopsis and conformance scope

The `cc` module specifies a portable surface for declarative C and C++ native builds. It dispatches on source-file extension (`.c`, `.C` → `cc`; `.cpp`, `.cc`, `.cxx`, `.C++` → `cxx`) so a single target may mix C and C++ sources without per-target language opt-in.

### Why `cc` rather than `cpp` (Standard Modules §9.2)

The blessed C-family build module is named `cc`, not `cpp`. The module handles `.c` and `.cpp` (plus `.cc`, `.cxx`) sources uniformly: extension dispatch chooses the appropriate compiler from a single toolchain triple. ### 9.2.3 Public surface

#### 9.2.3.1 cc.bin

```
cc.bin(name: string, opts: BinOpts) -> string
```

Registers an executable target named `name`. A conforming implementation MUST compile every source listed in `opts.sources` (or globbed from `opts.dir`) and link the resulting objects to `build/bin/<name>`. The function MUST register the target in the implementation's known-targets state so `cc.compile_commands()` (§9.2.3.11) can recover its inputs.

`BinOpts` fields (all optional unless noted):

| Field | Type | Semantics |
|---|---|---|
| `sources` | list[string] | Source paths or globs. REQUIRED unless `dir` is provided. |
| `dir` | string | Root for recursive source globbing (`*.c`, `*.cpp`, `*.cc`, `*.cxx`). |
| `includes` | list[string] | Private include dirs. |
| `export_includes` | list[string] | Public include dirs propagated to consumers via `cook.export`. |
| `defines` | list[string] | Macro names or `NAME=value`. |
| `system_libs` | list[string] | Passed to the linker as `-l<name>`. |
| `extra_cflags` | string | Raw flags appended at compile. |
| `extra_ldflags` | string | Raw flags appended at link. |
| `links` | list[string] | Other cc-registered targets to link against (transitive propagation per §9.2.4). |
| `standard` | string | e.g., `"c++17"`, `"c11"`. |
| `warnings` | string | `"default"` (`-Wall`), `"strict"` (`-Wall -Wextra -Wpedantic`), `"none"`, or raw flags. |

Return: the target name (echo for chaining).

Error: if `#sources == 0` after globbing, the implementation MUST raise `[cc.bin] no sources found for target '<name>'` at the caller's line.

#### 9.2.3.2 cc.lib

```
cc.lib(name: string, opts: LibOpts) -> string
```

Registers a static-library target. A conforming implementation MUST compile sources as for `cc.bin`, then archive the objects to `build/lib/lib<name>.a`. `LibOpts` is `BinOpts` minus link-time fields (`extra_ldflags`, `system_libs` apply at consumer link time, not at archive time, but MAY be recorded for propagation).

Error model identical to §9.2.3.1.

#### 9.2.3.3 cc.shared

```
cc.shared(name: string, opts: SharedOpts) -> string
```

Registers a shared-library target. A conforming implementation MUST compile sources with `-fPIC` and link the objects to `build/lib/lib<name>.<ext>`, where `<ext>` is `so` on Linux and macOS and `dll` on Windows. The linker MUST be invoked with `-shared` on Linux / `-dynamiclib` on macOS / `/DLL` on Windows.

`SharedOpts` fields are identical to `BinOpts`.

#### 9.2.3.4 cc.headers

```
cc.headers(name: string, opts: HeadersOpts) -> string
```

Registers a header-only library: a logical target that records exports for transitive propagation but emits no compile or link units. `HeadersOpts` typically carries `export_includes` and `defines`. `cc.headers` MUST register the target in known-targets but MUST NOT call `cook.add_unit(...)`.

#### 9.2.3.5 cc.compile

```
cc.compile(source: string, opts: CompileOpts) -> string
```

Compiles a single source file to an object. A conforming implementation MUST emit one work unit per call with `output = "build/obj/<target_name>/<stem>.o"` (where `<stem>` is the source's filename without extension and `<target_name>` is `opts.target_name` or `"default"`). The unit's `discovered_inputs` MUST point at a make-format depfile so header dependencies participate in incremental rebuilds.

Returns the output path.

#### 9.2.3.6 cc.archive

```
cc.archive(objects: list[string], output: string) -> string
```

Archives `objects` into a static library at `output` via `ar rcs`. The implementation MUST emit one work unit whose `inputs = objects`, `output = output`, `command = "ar rcs <output> <objects...>"`.

#### 9.2.3.7 cc.link

```
cc.link(objects: list[string], output: string, opts: LinkOpts) -> string
```

Links objects to `output`. `LinkOpts.system_libs` is appended as `-l<name>`; `LinkOpts.extra_ldflags` is appended verbatim; `LinkOpts.shared=true` adds the platform's shared-library flag (§9.2.3.3).

#### 9.2.3.8 cc.find

```
cc.find(name: string, opts: FindOpts?) -> FindResult
```

Discovers a package by name and returns a structured result record.

`FindOpts` (all optional):

| Field | Type | Semantics |
|---|---|---|
| `version` | string | A semver-style constraint (e.g., `">=4.0"`). Ignored in v0.1; honoured starting v0.2. |

`FindResult` (returned by every `cc.find` call):

| Field | Type | Semantics |
|---|---|---|
| `found` | boolean | `true` iff any strategy located the package. |
| `cflags` | string | The raw cflags string (verbatim from pkg-config in v0.1). |
| `libs` | string | The raw libs string. |

| `lib_dirs` | list[string] | Parsed `-L` paths. |
| `frameworks` | list[string] | Parsed `-framework` names (macOS). |

cc.defaults(opts: DefaultsOpts)

|---|---|---|
| `compiler` | string | The `cxx` driver to use (e.g., `"clang++"`); the `cc` driver is derived. |

### 9.2.4 Transitive propagation

Every cc target maker (`cc.bin`/`cc.lib`/`cc.shared`/`cc.headers`) MUST call `cook.export(name, info)` recording the target's outward-facing fields:

- `info.includes` — directories the consumer should add to `-I<dir>`
- `info.defines` — macros the consumer should add to `-D<def>`
- `info.system_libs` — libraries the consumer should link with `-l<name>`
- `info.extra_ldflags` — raw flags to append at consumer link time
- `info.links` — further transitive targets (forms the propagation graph)
- `info.compile_info` — per-target compile info for `cc.compile_commands` (§9.2.3.11)

When `opts.links = { "a", "b" }` is provided to a target maker, the implementation MUST resolve the closure by walking `cook.import("a").links`, `cook.import("b").links`, recursively, deduplicating both nodes (cycle-safe) and field values. The closure's `includes` / `defines` / `system_libs` / `extra_ldflags` MUST be merged into the new target's compile and link commands.

v0.1 normatively specifies implicit-public propagation: every field in the target maker's options is exported as-is. Future Standard revisions will introduce PRIVATE / PUBLIC distinctions; v0.1 implementations MAY anticipate the split by recognising `export_includes` (already a documented field on `BinOpts` per §9.2.3.1) as the public form; absence of `export_*` means the corresponding `includes`/etc. is exported by default.

### 9.2.5 Error model

Every error MUST be prefixed with `[cc.<fn>]` and named at level 2 (`error(msg, 2)`) so the blame surfaces at the Cookfile call site, not inside the module. Negative cases that MUST be diagnosed:

| Function | Condition | Message |
|---|---|---|
| `cc.bin`/`lib`/`shared` | `#sources == 0` after globbing | `no sources found for target '<name>'` |
| `cc.compile` | source file does not exist | `source file not found: <path>` |
| `cc.toolchain` (during `init()`) | no compiler on PATH | `no C/C++ compiler on PATH (tried g++, clang++)` |

`cc.find` does NOT raise on a miss; it returns `{ found = false }`. Callers that require a hit MUST test `result.found` and raise their own error.

### 9.2.6 Vendoring after install

Per §9.1.2, a project MAY hand-vendor a copy of `cook_cc` by placing `cook_modules/cook_cc.lua` (or `cook_modules/cook_cc/init.lua`) at the top of the project's `cook_modules/` directory. The hand-vendored file becomes the source of truth for that project; subsequent `cook modules install` invocations MUST leave it untouched (the install only writes under `cook_modules/share/lua/5.4/`).

A re-installed rock at `share/lua/5.4/cook_cc/init.lua` coexisting with a hand-vendored `cook_cc.lua` at the top level is unambiguously resolved: §7 specifies hand-vendored top-level wins.
